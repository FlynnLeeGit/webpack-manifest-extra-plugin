/**
 * get file extname 'a.js -> .js'
 * @param {* string }

const getExt = s => {
  const mapReg = /\.map$/
  if (mapReg.test(s)) {
    return (
      '.' +
      s
        .split('.')
        .slice(-2)
        .join('.')
    )
  }
  return path.extname(s)
}
